5 /*
6  FILE eiquadprog.hh
7 
8  NOTE: this is a modified of uQuadProg++ package, working with Eigen data structures.
9  uQuadProg++ is itself a port made by Angelo Furfaro of QuadProg++ originally developed by
10  Luca Di Gaspero, working with ublas data structures.
11 
12  The quadprog_solve() function implements the algorithm of Goldfarb and Idnani
13  for the solution of a (convex) Quadratic Programming problem
14 by means of a dual method.
15 
16 The problem is in the form:
17 
18 
19 min 0.5 * x G x + g0 x
20 s.t.
21  CE^T x + ce0 = 0
22  CI^T x + ci0 >= 0
23 
24  The matrix and vectors dimensions are as follows:
25  G: n * n
26  g0: n
27 
28  CE: n * p
29  ce0: p
30 
31  CI: n * m
32  ci0: m
33 
34  x: n
35 
36  The function will return the cost of the solution written in the x vector or
37  std::numeric_limits::infinity() if the problem is infeasible. In the latter case
38  the value of the x vector is not correct.
39 
40  References: D. Goldfarb, A. Idnani. A numerically stable dual method for solving
41  strictly convex quadratic programs. Mathematical Programming 27 (1983) pp. 1-33.
42 
43  Notes:
44  1. pay attention in setting up the vectors ce0 and ci0.
45  If the constraints of your problem are specified in the form
46  A^T x = b and C^T x >= d, then you should set ce0 = -b and ci0 = -d.
47  2. The matrix G is modified within the function since it is used to compute
48  the G = L^T L cholesky factorization for further computations inside the function.
49  If you need the original matrix G you should make a copy of it and pass the copy
50  to the function.

54  acknowledge the contribution of this modified function and of Di Gaspero's
55  original version in their research papers.

94 inline Scalar distance(Scalar a, Scalar b)
95 {
96  Scalar a1, b1, t;
97  a1 = std::abs(a);
98  b1 = std::abs(b);
99  if (a1 > b1)
100  {
101  t = (b1 / a1);
102  return a1 * std::sqrt(1.0 + t * t);
103  }
104  else
105  if (b1 > a1)
106  {
107  t = (a1 / b1);
108  return b1 * std::sqrt(1.0 + t * t);
109  }
110  return a1 * std::sqrt(2.0);
111 }

134 inline double solve_quadprog(MatrixXd & G, VectorXd & g0,
135  const MatrixXd & CE, const VectorXd & ce0,
136  const MatrixXd & CI, const VectorXd & ci0,
137  VectorXd& x, unsigned int maxIterations)
138 {
139  int i, j, k, l; /* indices */
140  int ip, me, mi;
141  int n=g0.size(); int p=ce0.size(); int m=ci0.size();
142  MatrixXd R(G.rows(),G.cols()), J(G.rows(),G.cols());
143 
144  LLT<MatrixXd,Lower> chol(G.cols());
145 
146  VectorXd s(m+p), z(n), r(m + p), d(n), np(n), u(m + p);
147  VectorXd x_old(n), u_old(m + p);
148  double f_value, psi, c1, c2, sum, ss, R_norm;
149  const double inf = std::numeric_limits<double>::infinity();
150  double t, t1, t2; /* t is the step length, which is the minimum of the partial step length t1
151  * and the full step length t2 */
152  VectorXi A(m + p), A_old(m + p), iai(m + p);
153  int q;
154  int iq, iter = 0;
155  bool iaexcl[m + p];
156 
157  me = p; /* number of equality constraints */
158  mi = m; /* number of inequality constraints */
159  q = 0; /* size of the active set A (containing the indices of the active constraints) */
160 
161  /*
162  * Preprocessing phase
163  */
164 
165  /* compute the trace of the original matrix G */
166  c1 = G.trace();
167 
168  /* decompose the matrix G in the form LL^T */
169  chol.compute(G);
170 
171  /* initialize the matrix R */
172  d.setZero();
173  R.setZero();
174  R_norm = 1.0; /* this variable will hold the norm of the matrix R */
175 
176  /* compute the inverse of the factorized matrix G^-1, this is the initial value for H */
177  // J = L^-T
178  J.setIdentity();
179  J = chol.matrixU().solve(J);
180  c2 = J.trace();
181 #ifdef TRACE_SOLVER
182  print_matrix("J", J, n);
183 #endif
184 
185  /* c1 * c2 is an estimate for cond(G) */
186 
187  /*
188  * Find the unconstrained minimizer of the quadratic form 0.5 * x G x + g0 x
189  * this is a feasible point in the dual space
190  * x = G^-1 * g0
191  */
192  x = chol.solve(g0);
193  x = -x;
194  /* and compute the current solution value */
195  f_value = 0.5 * g0.dot(x);
196 #ifdef TRACE_SOLVER
197  std::cerr << "Unconstrained solution: " << f_value << std::endl;
198  print_vector("x", x, n);
199 #endif
200 
201  /* Add equality constraints to the working set A */
202  iq = 0;
203  for (i = 0; i < me; i++)
204  {
205  np = CE.col(i);
206  compute_d(d, J, np);
207  update_z(z, J, d, iq);
208  update_r(R, r, d, iq);
209 #ifdef TRACE_SOLVER
210  print_matrix("R", R, iq);
211  print_vector("z", z, n);
212  print_vector("r", r, iq);
213  print_vector("d", d, n);
214 #endif
215 
216  /* compute full step length t2: i.e., the minimum step in primal space s.t. the contraint
217  becomes feasible */
218  t2 = 0.0;
219  if (std::abs(z.dot(z)) > std::numeric_limits<double>::epsilon()) // i.e. z != 0
220  t2 = (-np.dot(x) - ce0(i)) / z.dot(np);
221 
222  x += t2 * z;
223 
224  /* set u = u+ */
225  u(iq) = t2;
226  u.head(iq) -= t2 * r.head(iq);
227 
228  /* compute the new solution value */
229  f_value += 0.5 * (t2 * t2) * z.dot(np);
230  A(i) = -i - 1;
231 
232  if (!add_constraint(R, J, d, iq, R_norm))
233  {
234  // FIXME: it should raise an error
235  // Equality constraints are linearly dependent
236  return f_value;
237  }
238  }
239 
240  /* set iai = K \ A */
241  for (i = 0; i < mi; i++)
242  iai(i) = i;
243 
244 l1: iter++;
245  if(iter>maxIterations+1) return f_value;
246 #ifdef TRACE_SOLVER
247  print_vector("x", x, n);
248 #endif
249  /* step 1: choose a violated constraint */
250  for (i = me; i < iq; i++)
251  {
252  ip = A(i);
253  iai(ip) = -1;
254  }
255 
256  /* compute s(x) = ci^T * x + ci0 for all elements of K \ A */
257  ss = 0.0;
258  psi = 0.0; /* this value will contain the sum of all infeasibilities */
259  ip = 0; /* ip will be the index of the chosen violated constraint */
260  for (i = 0; i < mi; i++)
261  {
262  iaexcl[i] = true;
263  sum = CI.col(i).dot(x) + ci0(i);
264  s(i) = sum;
265  psi += std::min(0.0, sum);
266  }
267 #ifdef TRACE_SOLVER
268  print_vector("s", s, mi);
269 #endif
270 
271 
272  if (std::abs(psi) <= mi * std::numeric_limits<double>::epsilon() * c1 * c2* 100.0)
273  {
274  /* numerically there are not infeasibilities anymore */
275  q = iq;
276  return f_value;
277  }
278 
279  /* save old values for u, x and A */
280  u_old.head(iq) = u.head(iq);
281  A_old.head(iq) = A.head(iq);
282  x_old = x;
283 
284 l2: /* Step 2: check for feasibility and determine a new S-pair */
285  for (i = 0; i < mi; i++)
286  {
287  if (s(i) < ss && iai(i) != -1 && iaexcl[i])
288  {
289  ss = s(i);
290  ip = i;
291  }
292  }
293  if (ss >= 0.0)
294  {
295  q = iq;
296  return f_value;
297  }
298 
299  /* set np = n(ip) */
300  np = CI.col(ip);
301  /* set u = (u 0)^T */
302  u(iq) = 0.0;
303  /* add ip to the active set A */
304  A(iq) = ip;
305 
306 #ifdef TRACE_SOLVER
307  std::cerr << "Trying with constraint " << ip << std::endl;
308  print_vector("np", np, n);
309 #endif
310 
311 l2a:/* Step 2a: determine step direction */
312  /* compute z = H np: the step direction in the primal space (through J, see the paper) */
313  compute_d(d, J, np);
314  update_z(z, J, d, iq);
315  /* compute N* np (if q > 0): the negative of the step direction in the dual space */
316  update_r(R, r, d, iq);
317 #ifdef TRACE_SOLVER
318  std::cerr << "Step direction z" << std::endl;
319  print_vector("z", z, n);
320  print_vector("r", r, iq + 1);
321  print_vector("u", u, iq + 1);
322  print_vector("d", d, n);
323  print_ivector("A", A, iq + 1);
324 #endif
325 
326  /* Step 2b: compute step length */
327  l = 0;
328  /* Compute t1: partial step length (maximum step in dual space without violating dual feasibility */
329  t1 = inf; /* +inf */
330  /* find the index l s.t. it reaches the minimum of u+(x) / r */
331  for (k = me; k < iq; k++)
332  {
333  double tmp;
334  if (r(k) > 0.0 && ((tmp = u(k) / r(k)) < t1) )
335  {
336  t1 = tmp;
337  l = A(k);
338  }
339  }
340  /* Compute t2: full step length (minimum step in primal space such that the constraint ip becomes feasible */
341  if (std::abs(z.dot(z)) > std::numeric_limits<double>::epsilon()) // i.e. z != 0
342  t2 = -s(ip) / z.dot(np);
343  else
344  t2 = inf; /* +inf */
345 
346  /* the step is chosen as the minimum of t1 and t2 */
347  t = std::min(t1, t2);
348 #ifdef TRACE_SOLVER
349  std::cerr << "Step sizes: " << t << " (t1 = " << t1 << ", t2 = " << t2 << ") ";
350 #endif
351 
352  /* Step 2c: determine new S-pair and take step: */
353 
354  /* case (i): no step in primal or dual space */
355  if (t >= inf)
356  {
357  /* QPP is infeasible */
358  // FIXME: unbounded to raise
359  q = iq;
360  return inf;
361  }
362  /* case (ii): step in dual space */
363  if (t2 >= inf)
364  {
365  /* set u = u + t * [-r 1) and drop constraint l from the active set A */
366  u.head(iq) -= t * r.head(iq);
367  u(iq) += t;
368  iai(l) = l;
369  delete_constraint(R, J, A, u, p, iq, l);
370 #ifdef TRACE_SOLVER
371  std::cerr << " in dual space: "
372  << f_value << std::endl;
373  print_vector("x", x, n);
374  print_vector("z", z, n);
375  print_ivector("A", A, iq + 1);
376 #endif
377  goto l2a;
378  }
379 
380  /* case (iii): step in primal and dual space */
381 
382  x += t * z;
383  /* update the solution value */
384  f_value += t * z.dot(np) * (0.5 * t + u(iq));
385 
386  u.head(iq) -= t * r.head(iq);
387  u(iq) += t;
388 #ifdef TRACE_SOLVER
389  std::cerr << " in both spaces: "
390  << f_value << std::endl;
391  print_vector("x", x, n);
392  print_vector("u", u, iq + 1);
393  print_vector("r", r, iq + 1);
394  print_ivector("A", A, iq + 1);
395 #endif
396 
397  if (t == t2)
398  {
399 #ifdef TRACE_SOLVER
400  std::cerr << "Full step has taken " << t << std::endl;
401  print_vector("x", x, n);
402 #endif
403  /* full step has taken */
404  /* add constraint ip to the active set*/
405  if (!add_constraint(R, J, d, iq, R_norm))
406  {
407  iaexcl[ip] = false;
408  delete_constraint(R, J, A, u, p, iq, ip);
409 #ifdef TRACE_SOLVER
410  print_matrix("R", R, n);
411  print_ivector("A", A, iq);
412 #endif
413  try
414  {
415  for (i = 0; i < m; i++)
416  iai(i) = i;
417  for (i = 0; i < iq; i++)
418  {
419  A(i) = A_old(i);
420  iai(A(i)) = -1;
421  u(i) = u_old(i);
422  }
423  x = x_old;
424  }
425  catch(std::exception e)
426  {
427  std::cout << e.what();
428  }
429 
430  goto l2; /* go to step 2 */
431  }
432  else
433  iai(ip) = -1;
434 #ifdef TRACE_SOLVER
435  print_matrix("R", R, n);
436  print_ivector("A", A, iq);
437 #endif
438  goto l1;
439  }
440 
441  /* a patial step has taken */
442 #ifdef TRACE_SOLVER
443  std::cerr << "Partial step has taken " << t << std::endl;
444  print_vector("x", x, n);
445 #endif
446  /* drop constraint l */
447  iai(l) = l;
448  delete_constraint(R, J, A, u, p, iq, l);
449 #ifdef TRACE_SOLVER
450  print_matrix("R", R, n);
451  print_ivector("A", A, iq);
452 #endif
453 
454  s(ip) = CI.col(ip).dot(x) + ci0(ip);
455 
456 #ifdef TRACE_SOLVER
457  print_vector("s", s, mi);
458 #endif
459  goto l2a;
460 }

463 inline bool add_constraint(MatrixXd& R, MatrixXd& J, VectorXd& d, int& iq, double& R_norm)
464 {
465  int n=J.rows();
466 #ifdef TRACE_SOLVER
467  std::cerr << "Add constraint " << iq << '/';
468 #endif
469  int i, j, k;
470  double cc, ss, h, t1, t2, xny;
471 
472  /* we have to find the Givens rotation which will reduce the element
473  d(j) to zero.
474  if it is already zero we don't have to do anything, except of
475  decreasing j */
476  for (j = n - 1; j >= iq + 1; j--)
477  {
478  /* The Givens rotation is done with the matrix (cc cs, cs -cc).
479  If cc is one, then element (j) of d is zero compared with element
480  (j - 1). Hence we don't have to do anything.
481  If cc is zero, then we just have to switch column (j) and column (j - 1)
482  of J. Since we only switch columns in J, we have to be careful how we
483  update d depending on the sign of gs.
484  Otherwise we have to apply the Givens rotation to these columns.
485  The i - 1 element of d has to be updated to h. */
486  cc = d(j - 1);
487  ss = d(j);
488  h = distance(cc, ss);
489  if (h == 0.0)
490  continue;
491  d(j) = 0.0;
492  ss = ss / h;
493  cc = cc / h;
494  if (cc < 0.0)
495  {
496  cc = -cc;
497  ss = -ss;
498  d(j - 1) = -h;
499  }
500  else
501  d(j - 1) = h;
502  xny = ss / (1.0 + cc);
503  for (k = 0; k < n; k++)
504  {
505  t1 = J(k,j - 1);
506  t2 = J(k,j);
507  J(k,j - 1) = t1 * cc + t2 * ss;
508  J(k,j) = xny * (t1 + J(k,j - 1)) - t2;
509  }
510  }
511  /* update the number of constraints added*/
512  iq++;
513  /* To update R we have to put the iq components of the d vector
514  into column iq - 1 of R
515  */
516  R.col(iq-1).head(iq) = d.head(iq);
517 #ifdef TRACE_SOLVER
518  std::cerr << iq << std::endl;
519 #endif
520 
521  if (std::abs(d(iq - 1)) <= std::numeric_limits<double>::epsilon() * R_norm)
522  // problem degenerate
523  return false;
524  R_norm = std::max<double>(R_norm, std::abs(d(iq - 1)));
525  return true;
526 }

529 inline void delete_constraint(MatrixXd& R, MatrixXd& J, VectorXi& A, VectorXd& u, int p, int& iq, int l)
530 {
531 
532  int n = R.rows();
533 #ifdef TRACE_SOLVER
534  std::cerr << "Delete constraint " << l << ' ' << iq;
535 #endif
536  int i, j, k, qq;
537  double cc, ss, h, xny, t1, t2;
538 
539  /* Find the index qq for active constraint l to be removed */
540  for (i = p; i < iq; i++)
541  if (A(i) == l)
542  {
543  qq = i;
544  break;
545  }
546 
547  /* remove the constraint from the active set and the duals */
548  for (i = qq; i < iq - 1; i++)
549  {
550  A(i) = A(i + 1);
551  u(i) = u(i + 1);
552  R.col(i) = R.col(i+1);
553  }
554 
555  A(iq - 1) = A(iq);
556  u(iq - 1) = u(iq);
557  A(iq) = 0;
558  u(iq) = 0.0;
559  for (j = 0; j < iq; j++)
560  R(j,iq - 1) = 0.0;
561  /* constraint has been fully removed */
562  iq--;
563 #ifdef TRACE_SOLVER
564  std::cerr << '/' << iq << std::endl;
565 #endif
566 
567  if (iq == 0)
568  return;
569 
570  for (j = qq; j < iq; j++)
571  {
572  cc = R(j,j);
573  ss = R(j + 1,j);
574  h = distance(cc, ss);
575  if (h == 0.0)
576  continue;
577  cc = cc / h;
578  ss = ss / h;
579  R(j + 1,j) = 0.0;
580  if (cc < 0.0)
581  {
582  R(j,j) = -h;
583  cc = -cc;
584  ss = -ss;
585  }
586  else
587  R(j,j) = h;
588 
589  xny = ss / (1.0 + cc);
590  for (k = j + 1; k < iq; k++)
591  {
592  t1 = R(j,k);
593  t2 = R(j + 1,k);
594  R(j,k) = t1 * cc + t2 * ss;
595  R(j + 1,k) = xny * (t1 + R(j,k)) - t2;
596  }
597  for (k = 0; k < n; k++)
598  {
599  t1 = J(k,j);
600  t2 = J(k,j + 1);
601  J(k,j) = t1 * cc + t2 * ss;
602  J(k,j + 1) = xny * (J(k,j) + t1) - t2;
603  }
604  }
605 }
